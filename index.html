    var app = angular.module('Mycalc', []);
    app.controller('myctrl', function ($scope) {


        $scope.calc = function () {
            var n1 = parseInt($scope.num1);
            var n2 = parseInt($scope.num2);
            var op=['Add','Sub','Mul','Div'];
            switch ($scope.op) {
                case 'Add': $scope.result = n1 + n2;
                    break;
                case 'Sub': $scope.result = n1 - n2;
                    break;
                case 'Mul': $scope.result = n1 * n2;
                    break;
                case 'Div': $scope.result = n1 / n2;
                    break;
                
                default:alert("Please Select the operation!!....")
            }
        };

    });
</script>
